<body>

  <div id="chat-container">

    <div id="chat-box"></div>

    <div id="input-row">

      <input id="user-input" type="text" placeholder="Écris ton message...">

      <button onclick="sendMessage()">Envoyer</button>

    </div>

  </div>


  <script>

    const chatBox = document.getElementById("chat-box");

    const input = document.getElementById("user-input");


    // A1-level conversation prompts

    const prompts = [

      "Bonjour ! Comment ça va ?",

      "Quel est ton prénom ?",

      "Tu as quel âge ?",

      "Où habites-tu ?",

      "Qu’est-ce que tu aimes manger ?",

      "Est-ce que tu as des frères ou des sœurs ?"

    ];


    function addMessage(sender, text) {

      const msg = document.createElement("div");

      msg.className = sender;

      msg.textContent = (sender === "user" ? "👤 " : "🤖 ") + text;

      chatBox.appendChild(msg);

      chatBox.scrollTop = chatBox.scrollHeight;

    }


    function sendMessage() {

      const userText = input.value.trim();

      if (!userText) return;

      addMessage("user", userText);

      input.value = "";


      // Bot picks a random prompt

      setTimeout(() => {

        const botText = prompts[Math.floor(Math.random() * prompts.length)];

        addMessage("bot", botText);

      }, 500);

    }

  </script>
